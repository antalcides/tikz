  \tkzInit[xmax=5,ymax=5]  \tkzGrid
  \tkzDefPoint(3,2){A}  \tkzDefPoint(1,1){B}
  \tkzCalcLength[cm](A,B) \tkzGetLength{rAB}
  \tkzDrawCircle[R](A,\rAB cm)
  \tkzGetRandPointOn[circle = center A radius \rAB cm]{a}
  \tkzDrawSegment(A,a)
  \tkzDrawPoints(A,B,a)
  \tkzLabelPoints(A,B,a)
\end{tikzpicture}
